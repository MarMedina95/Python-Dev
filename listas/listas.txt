>>> n.append(30)
>>> n
[23, 34, 12, 54, 67, 30]
------------------------------------------------------------
>>> max(n) // nos dice el numero mayor de la cadena
67
-----------------------------------------------------------
>>> min(n) // el menor de la cadena
12
------------------------------------------------------------
>>> n.remove(12)  // remueve ese elemento de la cadena
>>> n
[23, 34, 54, 67, 30]
-----------------------------------------------------------
>>> n.reverse() // voltea el orden de la cadena
>>> n
[30, 67, 54, 34, 23]
---------------------------------------------------------
>>> n.sort()  // ordena la cadena de menor a mayor
>>> n
[23, 30, 34, 54, 67]
